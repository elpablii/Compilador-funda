// Ejemplo 1: Aritmética básica
enteros a;
flotantes b;
a = 5;
b = 2.5;
a = a + 3;
b = b * 2.0;
print(a);
print(b);

// Ejemplo 2: Lógica booleana y control
booleano bandera;
bandera = true;
if (bandera) {
    print("La bandera es verdadera");
} else {
    print("La bandera es falsa");
}

// Ejemplo 3: Operaciones lógicas
booleano x;
booleano y;
x = true;
y = false;
booleano resultado;
resultado = x && y;
print(resultado);
resultado = x || y;
print(resultado);
resultado = !x;
print(resultado);

// Ejemplo 4: Comparaciones y ciclo
enteros i;
i = 0;
while (i < 3) {
    print(i);
    i = i + 1;
}

// Ejemplo 5: Combinación de lógica y aritmética
enteros num;
booleano esMayor;
num = 10;
esMayor = num > 5;
if (esMayor) {
    print("num es mayor que 5");
}
